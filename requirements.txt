# AI Browser Controller — Python dependencies
#
# Install ONLY inside a virtualenv (never globally):
#     python3.12 -m venv venv
#     source venv/bin/activate          # macOS/Linux
#     venv\Scripts\activate             # Windows
#     pip install -r requirements.txt
#
# AFTER pip install, download the browser binaries (required, one-time):
#     playwright install chromium
#
# SYSTEM dependency (NOT a pip package): ffmpeg must be installed separately and
# be on your PATH for screen recording / MP4 encoding to work. Install it with:
#     macOS:        brew install ffmpeg
#     Debian/Ubuntu: sudo apt install ffmpeg
#     Windows:      download from https://www.gyan.dev/ffmpeg/builds/ (or `winget install ffmpeg`)

# ----- core -----
pydantic>=2.6            # v2 — settings/command/response models

# ----- browser -----
playwright>=1.42         # browser automation engine (run `playwright install chromium`)

# ----- api -----
fastapi>=0.110           # HTTP + WebSocket server surface
uvicorn[standard]>=0.29  # ASGI server (standard extras: websockets, httptools, uvloop)
websockets>=12.0         # WebSocket client/server protocol

# ----- vision -----
opencv-python>=4.9       # frame processing / video assembly helpers
pillow>=10.2             # image loading, annotation, resizing
numpy>=1.26              # array math backing the vision pipeline

# ----- ai -----
mcp>=1.2                 # Model Context Protocol stdio server (FastMCP)
anthropic>=0.45          # Claude adapter (default model: claude-opus-4-8)
openai>=1.30             # OpenAI adapter

# ----- no-image mode -----
# Microsoft's MarkItDown converts images/PDF/Office/HTML to markdown so an agent
# can read media as text (the `to_markdown` tool + global no-image mode). Optional
# at runtime: the media helpers degrade gracefully if it is not installed.
# The [pdf,docx,pptx,xlsx] extras pull the per-format backends — without them
# base MarkItDown only handles HTML/text/CSV/JSON (PDF would raise
# MissingDependency). `set_no_image_mode`/`to_markdown` report which formats work.
markitdown[pdf,docx,pptx,xlsx]>=0.0.1  # image/PDF/Office -> markdown (no-image mode)

# ----- OCR (read text baked into images/screenshots) -----
# Optional at runtime: `extract_text_from_screenshot` / `read_image` degrade to a
# clear error if missing. Needs BOTH this package AND the system Tesseract binary
# (e.g. `brew install tesseract` / `apt-get install tesseract-ocr`).
pytesseract>=0.3.10  # optional OCR engine wrapper

# ----- test -----
pytest>=8.0              # test runner
pytest-asyncio>=0.23     # async test support (asyncio_mode=auto, see pyproject.toml)
httpx>=0.27              # async HTTP client used by FastAPI TestClient and tests
